# Regex to capture street names that have a number in them
# Checks to ensure we are not capturing the numbers that are part of a date or medical measurements
(?i)(?<!(\/))(?<!(\d{2}\-))(?<!(\stype\s))(?!\d+\s(mg|units|year|years|yrs|qd|bpm|refills|tab|tabs|day|days|cc|ccs|mmhg|po|puff|mm|amp)\s)\b((\d+)\s{1,4}(?!Dear\b)([A-Z]\s{1,4}[A-Z][a-zA-Z]+|[A-Z][a-zA-Z]+\s{1,4}[A-Z][a-zA-Z]+|[A-Z][a-zA-Z]+|[A-Z]|("""+day_numbering+r"""))|((?<=on\s|up\s)\d{1,3}(st|nd|rd|th)))\s{1,4}(("""+address_indicator+r""")|[A-Z0-9]\slevel)(\.)?(\s#\d+)?(\,\s[a-z]+\sunit)?\b
